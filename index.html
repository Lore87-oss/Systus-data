// ------- AREE VERDI (GeoJSON) ---------- fetch("https://lore87-oss.github.io/Systus-data/Aree_verdi.geojson") .then(response => response.json()) .then(data => { L.geoJSON(data, { style: { color: "green", weight: 2, fillColor: "lightgreen", fillOpacity: 0.25 }, onEachFeature: function (feature, layer) { const p = feature.properties; layer.bindPopup(` ${p.Nome || "Area verde"}
Indirizzo: ${p.Indirizzo || "N/D"}
Tipologia gestione: ${p.Tipologia_Gestione || "N/D"}
Stato manutentivo: ${p.Stato_Manutentivo || "N/D"}
Alberi: ${p.Alberi || "N/D"} `); } }).addTo(map); });
